## 5. CHANGE B — Make the Routine page editable (add / edit / remove / reorder)

Today `#tab-routine` is read-only (`renderRoutines` only lets you check steps off). Make it a
small editor. **Scope cut to keep usage low: routine create/edit/remove/reorder are
LIVE-ONLY.** If `app.mode!=="live"`, `toast("Connect to edit routines")` and return — do NOT

today). This is deliberate: routine editing is infrequent config, not daily flow.

### 5a. Pure helpers in `cc-data.js` (TDD first)

```js
// field ∈ {name,why,mins,when,order}; returns the Notion properties patch for one edit
function routinePropsFor(field, value){
  if(field==="name") return {Routine:String(value)};
  if(field==="why")  return {Why:String(value)};
  if(field==="mins") return {Mins:(+value||0)};
  if(field==="when") return {"Time Of Day":String(value)};
  if(field==="order")return {Order:(+value||0)};
  return {};
}
// properties for a brand-new routine page
function newRoutineProps(r){
  return {Routine:String(r.name||""),Active:"__YES__","Done Today":"__NO__",
    "Time Of Day":String(r.when||"Morning"),Mins:(+r.mins||0),Why:String(r.why||""),Order:(+r.order||99)};
}
// swap one routine's Order with its neighbor in direction dir ∈ {-1,+1}
// returns [{id,order},...] writes needed (empty if no move possible). Pure.
function reorderSwap(routines, id, dir){
  var s=routines.slice().sort(function(a,b){return a.order-b.order;});
  var i=s.findIndex(function(r){return r.id===id;});
  var j=i+dir; if(i<0||j<0||j>=s.length) return [];
  var a=s[i],b=s[j];
  return [{id:a.id,order:b.order},{id:b.id,order:a.order}];
}
```
Export all three in the module's return object. Add tests covering each field map, the new-page
shape, and reorder at top/middle/bottom (bottom+1 and top-1 return `[]`).

### 5c. Actions (live-only; optimistic local update + one Notion write each)

```js
async function editRoutine(id, field, value){
  if(app.mode!=="live") return toast("Connect to edit routines");
  var r=app.routines.filter(function(x){return x.id===id;})[0]; if(!r) return;
  if(field==="name")r.name=value; else if(field==="why")r.why=value;
  else if(field==="mins")r.mins=+value||0; else if(field==="when")r.when=value;
  else if(field==="order")r.order=+value||0;
  renderRoutineEditor(); renderSteps(); renderTopStats();
  try{ await call(T.update,{page_id:id,command:"update_properties",properties:CCData.routinePropsFor(field,value)}); }
  catch(e){ toast("Save failed"); }
}
async function addRoutine(r){
  if(app.mode!=="live") return toast("Connect to edit routines");
  try{
    var res=await call(T.create,{parent:{data_source_id:DBS.routines},pages:[{properties:CCData.newRoutineProps(r)}]});
    var o=toObj(res), id=o&&o.pages&&o.pages[0]&&o.pages[0].id;
    app.routines.push({id:id||("tmp-"+Date.now()),name:r.name,why:r.why||"",mins:+r.mins||0,when:r.when||"Morning",order:+r.order||99,done:false,active:true});
    app.routines.sort(function(a,b){return a.order-b.order;});
    renderRoutineEditor(); renderSteps(); renderTopStats(); toast("Routine added");
  }catch(e){ toast("Add failed"); }
}
async function removeRoutine(id){            // deactivate, don't delete
  if(app.mode!=="live") return toast("Connect to edit routines");
  app.routines=app.routines.filter(function(x){return x.id!==id;});
  renderRoutineEditor(); renderSteps(); renderTopStats(); toast("Routine removed");
  try{ await call(T.update,{page_id:id,command:"update_properties",properties:{Active:"__NO__"}}); }catch(e){}
}
async function moveRoutine(id, dir){
  if(app.mode!=="live") return toast("Connect to edit routines");
  var writes=CCData.reorderSwap(app.routines,id,dir); if(!writes.length) return;
  writes.forEach(function(w){var r=app.routines.filter(function(x){return x.id===w.id;})[0]; if(r)r.order=w.order;});
  app.routines.sort(function(a,b){return a.order-b.order;});
  renderRoutineEditor(); renderSteps();
  try{ for(var i=0;i<writes.length;i++){ await call(T.update,{page_id:writes[i].id,command:"update_properties",properties:{Order:writes[i].order}}); } }catch(e){}
}
```

### 6b. `groupTasksByArea` in `cc-data.js` (TDD)
Sort areas by **open (unfinished) task count, descending** so the busiest area is first and
empty areas sink to the bottom. Ties break by total count, then by the fixed `AREAS` order
(stable):
```js
function groupTasksByArea(tasks, areas){
  var groups=areas.map(function(area,i){
    var list=(tasks||[]).filter(function(t){return (t.area||"")===area;});
    var open=list.filter(function(t){return !t.done;}).length;
    return {area:area, tasks:list, open:open, _i:i};
  });
  groups.sort(function(a,b){ return (b.open-a.open) || (b.tasks.length-a.tasks.length) || (a._i-b._i); });
  return groups;
}
```
Export + test: all 7 areas present even when empty; tasks land in the right bucket; busiest-open
area first; empty areas last; equal-open areas keep the fixed `AREAS` order.
